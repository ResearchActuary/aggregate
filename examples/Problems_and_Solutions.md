```python
def qd(df, caption=''):
    f3 = lambda x: f'{x:.3f}'
    f4 = lambda x: f'{x:.4f}'
    g3 = lambda x: f'{x:.3g}'
    fc1 = lambda x: f'{x:,.1f}'
    fc0 = lambda x: f'{x:,.0f}'
    mns = df.mean()
    ans = {}
    for c, mn in zip(df, mns): 
        ans[c] = f4 if mn < 10 else fc0
    if caption == '': 
        display(df.style.format(ans))
    else:
        display(df.style.format(ans).set_caption(caption))
```
